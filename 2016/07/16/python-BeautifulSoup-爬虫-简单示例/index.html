python+BeautifulSoup 爬虫 正则 简单示例

  • BeautifulSoup
  • python
  • 正则
  • 爬虫

##这里是直接用源码+重要注释

#-*- coding:utf-8 -*-
from urllib import urlopen
from bs4 import BeautifulSoup
import requests
import re
import time

url = ""
domain = ""

sessions = requests.Session()
#注意; 这里没有对编码处理，输出中文字符乱码。。decode（“utf-8”）

#还有别的header也可以加
header = {"User-Agent":"Mozilla/5.0 (Macintosh; Intel Mac OS X 10_10_4) AppleWebKit/600.7.12 (KHTML, like Gecko) Version/8.0.7 Safari/600.7.12"}
req = sessions.get(url,headers=header)

bsObj = BeautifulSoup(req.text)
#可以直接找tbody，省的再先找table，但是这样没有那样稳妥
#findAll和find函数都是的第一个都是标签的类型，如p，div，后面的{}是属性集合
#第一个标签类型也可以不填，那就都找
table_tbody = bsObj.find("tbody",{"style":"table-layout:fixed;"})
trs = table_tbody.findAll("tr",{"class":"tr3 t_one"})

subURLs = []
for tr in trs:
_url = tr.find("a",{"target":"_blank"}).attrs["href”]  #找到标签后用attrs可以找到属性
subURLs.append(domain+_url)

#print '\n'.join(subURLs)
#边爬数据边写，避免中途异常后无数据
out = open("./out","w")

real_url = []
for _url in subURLs:
#记得异常处理
try:
    #这里避免请求太频繁，因此每次暂停2秒
    time.sleep(2)
    print 'In url: '+_url
    _req = sessions.get(_url,headers= header)
    _bsObj = BeautifulSoup(_req.text)
    _a = _bsObj.find("div",{"class":"tpc_content do_not_catch"}).find("a",{"style":"cursor:pointer"})
#_a.get_text() 可以得到a标签的文本
#（.*?）是正则里面的东西，可以匹配全部的内容，前面\’是’转义，此处我们的待匹配字符串是.src=.....’” style..
#我们要取到的是src后面...
    #real_url.append(re.findall(r'.src=\'(.*?)\'\" style',str(_a))[0])  #转成字符串str处理
    out.write(re.findall(r'.src=\'(.*?)\'\" style',str(_a))[0]+'\n')
except:
    print '爬取失败'
finally:
    out.close()

out.close()
#out.write('\n'.join(real_url))
